1737  DomainInsideSignature getDomainInside() const{
1738  auto rule = getRule();
1739  if ((rule == TasGrid::rule_gausshermite) || (rule == TasGrid::rule_gausshermiteodd)){ // unbounded domain
1740  return [](std::vector<double> const &)->bool{ return true; };
1741  }else if ((rule == TasGrid::rule_gausslaguerre) || (rule == TasGrid::rule_gausslaguerreodd)){ // bounded from below
1742  if (domain_transform_a.empty()){
1743  return [](std::vector<double> const &x)->bool{
1744  for(auto const &v : x) if (v < 0.0) return false;
1745  return true;
1746  };
1747  }else{
1748  size_t dims = (size_t) getNumDimensions();
1749  return [=](std::vector<double> const &x)->bool{
1750  for(size_t i=0; i<dims; i++) if (x[i] < domain_transform_a[i]) return false;
1751  return true;
1752  };
1753  }
1754  }else{
1755  if (domain_transform_a.empty()){
1756  if (isFourier()){
1757  return [](std::vector<double> const &x)->bool{
1758  for(auto const &v : x) if ((v < 0.0) || (v > 1.0)) return false;
1759  return true;
1760  };
1761  }else{
1762  return [](std::vector<double> const &x)->bool{
1763  for(auto const &v : x) if ((v < -1.0) || (v > 1.0)) return false;
1764  return true;
1765  };
1766  }
1767  }else{
1768  size_t dims = (size_t) getNumDimensions();
1769  return [=](std::vector<double> const &x)->bool{
1770  for(size_t i=0; i<dims; i++)
1771  if ((x[i] < domain_transform_a[i]) || (x[i] > domain_transform_b[i])) return false;
1772  return true;
1773  };
1774  }
1775  }
