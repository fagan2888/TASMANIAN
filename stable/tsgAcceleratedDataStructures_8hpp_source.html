638  ChangeType favorSparse(bool favor){
639  AlgorithmPreference new_preference = [=]()->AlgorithmPreference{
640  if (favor){
641  return (algorithm_select == algorithm_dense) ? algorithm_autoselect : algorithm_sparse;
642  }else{
643  return (algorithm_select == algorithm_sparse) ? algorithm_autoselect : algorithm_dense;
644  }
645  }();
646  if (new_preference != algorithm_select){
647  algorithm_select = new_preference;
648  return change_sparse_dense;
649  }else{
650  return change_none;
651  }
652  }

673  ChangeType enable(TypeAcceleration acc, int new_gpu_id){
674  // get the effective new acceleration mode (use the fallback if acc is not enabled)
675  TypeAcceleration effective_acc = AccelerationMeta::getAvailableFallback(acc);
676  // if switching to a GPU mode, check if the device id is valid
677  if (AccelerationMeta::isAccTypeGPU(effective_acc) and ((new_gpu_id < 0 or new_gpu_id >= AccelerationMeta::getNumCudaDevices())))
678  throw std::runtime_error("Invalid CUDA device ID, see ./tasgrid -v for list of detected devices.");
679 
680  // assign the new values for the mode and device, but remember the current gpu state and check whether something changed
681  bool was_on_gpu = on_gpu();
682  ChangeType mode_change = (effective_acc == Utils::exchange(mode, effective_acc)) ? change_none : change_cpu_blas;
683  ChangeType device_change = (new_gpu_id == Utils::exchange(device, new_gpu_id)) ? change_none : change_gpu_device;
684 
685  if (AccelerationMeta::isAccTypeGPU(mode)){
686  // if the new mode is GPU-based, reset the engine and the handles (if already created)
687  engine = Utils::make_unique<GpuEngine>();
688  }else{
689  engine.reset();
690  }
691 
692  if (was_on_gpu){
693  // switching from gpu to gpu issues a change if the device is different
694  // switching from gpu to non-gpu acts as if the device has changed
695  return (on_gpu()) ? device_change : change_gpu_device;
696  }else{
697  // switching from non-gpu to gpu triggers gpu-enabled switch
698  // switching from non-gpu to non-gpu triggers blas change if the modes were different
699  return (on_gpu()) ? change_gpu_enabled : mode_change;
700  }
701  }
703  void setDevice() const{ AccelerationMeta::setDefaultCudaDevice(device); }
705  operator GpuEngine* () const{ return engine.get(); }
707  bool on_gpu() const{ return !!engine; }
708 
709  #ifdef Tasmanian_ENABLE_CUDA

714  }
